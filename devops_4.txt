
require 'config.php';

$task_id = isset($_GET['task_id']) ? intval($_GET['task_id']) : 0;

// جلب الكود من قاعدة البيانات
$code_result = $conn->query("SELECT * FROM codes WHERE task_id = $task_id ORDER BY created_at DESC LIMIT 1");
$code = $code_result ? $code_result->fetch_assoc() : null;
if ($code) {
    $code['code'] = htmlspecialchars_decode($code['code'], ENT_QUOTES);
}
?>
<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>Test Task</title>
    <link href="https://maxcdn.bootstrapcdn.com/bootstrap/4.5.2/css/bootstrap.min.css" rel="stylesheet">
    <link href="https://cdnjs.cloudflare.com/ajax/libs/font-awesome/5.15.3/css/all.min.css" rel="stylesheet">
</head>
<body>
<div class="container mt-5">
    <h2>Test Task</h2>
    <div id="codeSection">
        <?php if ($code): ?>
            <table class="table table-bordered">
                <thead>
                <tr>
                    <th>Code</th>
                    <th>Actions</th>
                </tr>
                </thead>
                <tbody>
                <tr>
                    <td>
                        <pre id="codeContent"><?php echo htmlspecialchars($code['code']); ?></pre>
                    </td>
                    <td>
                        <button class="btn btn-warning" onclick="testCode(<?php echo $task_id; ?>)">Test</button>
                        <button class="btn btn-success" onclick="publishCode(<?php echo $task_id; ?>)">Publish</button>
                    </td>
                </tr>
                <tr>
                    <td colspan="2">
                        <div id="phpTestResult"></div>
                    </td>
                </tr>
                </tbody>
            </table>
        <?php else: ?>
            <p>No code found for this task.</p>
        <?php endif; ?>
    </div>
    
</div>

<script src="https://code.jquery.com/jquery-3.5.1.min.js"></script>
<script src="https://cdn.jsdelivr.net/npm/@popperjs/core@2.5.4/dist/umd/popper.min.js"></script>
<script src="https://maxcdn.bootstrapcdn.com/bootstrap/4.5.2/js/bootstrap.min.js"></script>
<script>
    function testCode(taskId) {
        var code = $('#codeContent').text(); // الحصول على الكود من العنصر الذي يحمل id="codeContent"
        $.ajax({
            url: 'test_code.php',
            type: 'POST',
            data: { task_id: taskId, code: code }, // إرسال الكود كبيانات POST
            success: function(response) {
                $('#phpTestResult').html('<strong>PHP Test Result:</strong><pre>' + response + '</pre>');
            },
            error: function(xhr, status, error) {
                $('#phpTestResult').html('<div class="alert alert-danger">Error testing code: ' + error + '<br>Response: ' + xhr.responseText + '</div>');
            }
        });
    }



    function publishCode(taskId) {
        $.ajax({
            url: 'publish_code.php',
            type: 'POST',
            data: {task_id: taskId},
            success: function(response) {
                alert(response);
            },
            error: function() {
                alert('Error publishing code');
            }
        });
    }

    
</script>
